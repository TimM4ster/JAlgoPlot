import javafx.application.Application;
import javafx.scene.Group;
import javafx.scene.Scene;
import javafx.scene.shape.Circle;
import javafx.stage.Stage;

public class GraphExample extends Application {
    @Override
    public void start(Stage primaryStage) {
        // Create the nodes
        Circle node1 = new Circle(50, 50, 20);
        Circle node2 = new Circle(150, 50, 20);

        // Create the graph and add the edge
        Group root = new Group();
        Graph graph = new Graph(root);
        graph.addEdge(node1, node2);

        // Add the nodes to the group
        root.getChildren().add(node1);
        root.getChildren().add(node2);

        // Create the scene and add the group to it
        Scene scene = new Scene(root, 200, 100);

        // Set the stage properties and show the stage
        primaryStage.setTitle("Graph Example");
        primaryStage.setScene(scene);
        primaryStage.show();
    }

    public static void main(String[] args) {
        launch(args);
    }
}
