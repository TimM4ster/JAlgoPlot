import javafx.scene.Group;
import javafx.scene.shape.Circle;
import javafx.scene.shape.Line;
import javafx.scene.shape.Polygon;

public class Graph {
    private Group root;

    public Graph(Group root) {
        this.root = root;
    }

    public void addDirectedEdge(Circle node1, Circle node2) {
        // Calculate the endpoints of the edge
        double startX = node1.getCenterX() + node1.getRadius();
        double startY = node1.getCenterY();
        double endX = node2.getCenterX() - node2.getRadius();
        double endY = node2.getCenterY();

        // Create the edge
        Line edge = new Line(startX, startY, endX, endY);

        // Calculate the points for the arrowhead
        double arrowSize = 10;
        double arrowAngle = Math.atan2(endY - startY, endX - startX);
        double arrowX1 = endX + arrowSize * Math.cos(arrowAngle + Math.PI / 6);
        double arrowY1 = endY + arrowSize * Math.sin(arrowAngle + Math.PI / 6);
        double arrowX2 = endX + arrowSize * Math.cos(arrowAngle - Math.PI / 6);
        double arrowY2 = endY + arrowSize * Math.sin(arrowAngle - Math.PI / 6);

        // Create the arrowhead
        Polygon arrowhead = new Polygon(endX, endY, arrowX1, arrowY1, arrowX2, arrowY2);

        // Add the edge and arrowhead to the scene graph
        root.getChildren().add(edge);
        root.getChildren().add(arrowhead);
    }
}
