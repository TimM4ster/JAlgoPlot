import javafx.scene.Group;
import javafx.scene.shape.Circle;
import javafx.scene.shape.Line;

public class Graph {
    private Group root;

    public Graph(Group root) {
        this.root = root;
    }

    public void addEdge(Circle node1, Circle node2) {
        Line edge = new Line(node1.getCenterX() + node1.getRadius(), node1.getCenterY(),
                             node2.getCenterX() - node2.getRadius(), node2.getCenterY());
        root.getChildren().add(edge);
    }
}
